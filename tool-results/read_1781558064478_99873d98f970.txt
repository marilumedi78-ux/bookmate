   329тЖТ   329тЖТfunction darkenHex(hex: string, amount = 0.3): string {
   330тЖТ   330тЖТ  if (!hex || !hex.startsWith('#')) return '#1a5c54'
   331тЖТ   331тЖТ  const num = parseInt(hex.slice(1), 16)
   332тЖТ   332тЖТ  const r = Math.max(0, Math.round(((num >> 16) & 0xff) * (1 - amount)))
   333тЖТ   333тЖТ  const g = Math.max(0, Math.round(((num >> 8) & 0xff) * (1 - amount)))
   334тЖТ   334тЖТ  const b = Math.max(0, Math.round((num & 0xff) * (1 - amount)))
   335тЖТ   335тЖТ  return `#${((r << 16) | (g << 8) | b).toString(16).padStart(6, '0')}`
   336тЖТ   336тЖТ}
